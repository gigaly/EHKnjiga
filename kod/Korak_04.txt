Public Function ucitaj_jednojezicni_rjecnik(NazivDatoteke As String, NazivTablice As String) As Long
Dim fst As New ADODB.Stream, pom As String
Dim tc As DAO.Recordset
Dim p() As String, i As Long

fst.Charset = "utf-8"
fst.Open
fst.Type = adTypeText
fst.LoadFromFile CurrentProject.Path & "\txt\" & NazivDatoteke
fst.LineSeparator = adCRLF

Set tc = CodeDb.OpenRecordset(NazivTablice)
i = 0
Do While Not fst.EOS
  pom = fst.ReadText(-2)
  p = Split(pom, vbTab)
  i = i + 1
  If i Mod 1000 = 0 Then
    Debug.Print (i)
    DoEvents
  End If
  If UBound(p) = 3 Then
    tc.AddNew
    tc("ID") = i
    tc("Kosi oblik") = p(0)
    tc("Osnovni oblik") = p(1)
    tc("Oznaka") = p(2)
    tc("Vrsta rije�i") = p(3)
    tc.Update
  Else
    Debug.Print
    Debug.Print "Datoteka: " & NazivDatoteke & "    Pogre�ka u retku " & i & ": " & pom
  End If
Loop
End Function
